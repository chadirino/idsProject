package draft;

import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class Mgmt_Inv_Sys extends JFrame{
	JFrame jf1;
	JPanel jp1;
	JLabel jl1;
	static JTextField jtf1;
	static JButton jb1;
	
	JPanel jp2;
	JLabel jl2;
	static JTextField jtf2;
	static JButton jb2;
	
	public Mgmt_Inv_Sys() {
		
		jf1 = new JFrame();
		//Container
		Container cp = getContentPane();
		cp.setLayout(new BoxLayout(cp, BoxLayout.Y_AXIS));
		cp.setLayout(new BoxLayout(cp, BoxLayout.Y_AXIS));
		
		//Panel 1
		jl1 = new JLabel("Add inventory item");
		jtf1 = new JTextField();
		jtf1.setPreferredSize(new Dimension(100,20));
		jp1 = new JPanel();
		jp1.setLayout(new BoxLayout(jp1, BoxLayout.Y_AXIS));
		
		//Panel 2
		jl2 = new JLabel("Add quantity");
		jtf2 = new JTextField();
		jtf2.setPreferredSize(new Dimension(100,20));
		jb1 = new JButton("+");
		jb2 = new JButton("-");
		jp2 = new JPanel();
		jp2.setLayout(new BoxLayout(jp2, BoxLayout.Y_AXIS));
		
		//add section
		cp.add(jp1);
		cp.add(jp2);
		
		jp1.add(jl1);
		jp1.add(jtf1);
		
		jp2.add(jl2);
		jp2.add(jtf2);
		jp2.add(jb1);
		jp2.add(jb2);
		
		//main GUI controls
		pack();
		jf1.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
		setVisible(true);
		Dimension dim = Toolkit.getDefaultToolkit().getScreenSize();
		this.setLocation(dim.width/2-this.getSize().width/2, dim.height/2-this.getSize().height/2);
	}
	
	public static void main(String[] args) {
		// TODO Auto-generated method stub

		new Mgmt_Inv_Sys();
	}
}
